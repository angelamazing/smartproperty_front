🔧 持续加载问题修复测试

测试修复后的第一次点击预约界面问题

问题分析

持续存在的问题

现象: 第一次点击预约界面仍然有加载失败问题

可能原因:

  • 页面生命周期冲突：onLoad() 和 onShow() 同时触发
  • 重复加载：onShow() 在页面初始化时也会被调用
  • 数据竞争：多个异步操作同时进行
  • 错误处理不够健壮

修复方案

1. 修复页面生命周期冲突

// 添加初始化标志
data() {
  return {
    isPageInitialized: false,
    // ... 其他数据
  }
}

onLoad() {
  this.initPage()
},

onShow() {
  // 只有在页面已经初始化后才刷新数据，避免重复加载
  if (this.isPageInitialized) {
    this.refreshData()
  }
}

2. 改进初始化流程

async initPage() {
  try {
    this.isLoading = true
    this.isPageInitialized = false
    
    // 设置默认日期
    const today = this.$getCurrentDate()
    this.selectedDate = this.$formatTime(today, 'YYYY-MM-DD')
    
    // 加载初始数据
    await this.loadInitialData()
    
    // 标记页面初始化完成
    this.isPageInitialized = true
    console.log('页面初始化完成')
  } catch (error) {
    console.error('页面初始化失败:', error)
    // 即使失败也标记为已初始化，避免重复尝试
    this.isPageInitialized = true
  } finally {
    this.isLoading = false
  }
}

3. 添加重试机制

// 在 loadInitialData 中添加重试逻辑
if (!hasAnyData) {
  console.warn('所有数据加载都失败了，尝试重试...')
  
  try {
    console.log('开始重试加载数据...')
    await this.retryLoadData()
    
    // 再次检查数据
    const retryHasAnyData = (this.venues?.length > 0) || 
                           (this.reservationRecords?.length > 0) || 
                           (this.scheduleData?.length > 0)
    
    if (!retryHasAnyData) {
      uni.showToast({
        title: '数据加载失败，请检查网络连接',
        icon: 'none'
      })
    } else {
      console.log('重试成功，数据已加载')
    }
  } catch (retryError) {
    console.error('重试也失败了:', retryError)
    uni.showToast({
      title: '数据加载失败，请检查网络连接',
      icon: 'none'
    })
  }
}

4. 添加重试加载方法

async retryLoadData() {
  try {
    console.log('重试加载数据...')
    
    // 使用更简单的顺序加载方式
    const venues = await this.loadVenues().catch(error => {
      console.warn('重试加载场地失败:', error)
      return []
    })
    
    const records = await this.loadReservationRecords().catch(error => {
      console.warn('重试加载预约记录失败:', error)
      return []
    })
    
    const schedule = await this.loadScheduleData().catch(error => {
      console.warn('重试加载安排数据失败:', error)
      return []
    })
    
    // 更新数据
    if (venues && venues.length > 0) this.venues = venues
    if (records && records.length > 0) this.reservationRecords = records
    if (schedule && schedule.length > 0) this.scheduleData = schedule
    
    return { venues, records, schedule }
  } catch (error) {
    console.error('重试加载数据失败:', error)
    throw error
  }
}

页面生命周期演示

修复前的生命周期

用户点击预约界面
    ↓
onLoad() 触发 → initPage() → loadInitialData()
    ↓
onShow() 触发 → refreshData() → loadVenues() (只加载当前标签页)
    ↓
结果：数据不完整，可能显示加载失败

修复后的生命周期

用户点击预约界面
    ↓
onLoad() 触发 → initPage() → loadInitialData() → isPageInitialized = true
    ↓
onShow() 触发 → 检查 isPageInitialized → 跳过 refreshData()
    ↓
结果：数据完整，页面正常显示
